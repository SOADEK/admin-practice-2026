Базовые команды Linux
ls — показать содержимое папки
ls -l — подробный список (права, владелец, размер, дата)
ls -a — показать скрытые файлы (начинаются с .)
ls -lh — «человекочитаемый» размер файлов (в КБ, МБ)
cd — перейти в другую папку
cd .. — подняться на уровень выше
cd ~ или просто cd — перейти в домашнюю директорию
cd - — вернуться в предыдущую папку
pwd — показать текущий путь (полный путь к рабочей директории)
mkdir — создать папку
mkdir -p project/logs — создать вложенную структуру (все промежуточные папки)
rm — удалить файл или папку
rm file.txt — удалить файл
rm -r folder/ — удалить папку рекурсивно
rm -rf folder/ — опасно! — принудительно удалить всё без подтверждения
cp — скопировать файл или папку
cp file.txt backup/ — скопировать файл в папку
cp -r source/ dest/ — скопировать папку рекурсивно
mv — переместить или переименовать
mv old.txt new.txt — переименовать файл
mv file.txt ~/Documents/ — переместить файл в другую папку
touch — создать пустой файл или обновить время изменения существующего
touch log.txt
cat — вывести содержимое файла в терминал
cat config.txt
less — просмотреть файл постранично (удобно для больших файлов)
less /var/log/syslog → навигация: стрелки, q — выйти
head / tail — показать начало или конец файла
tail -f /var/log/nginx/access.log — живой просмотр лога (полезно при отладке)
nano / vim — текстовые редакторы в терминале
nano script.sh — простой редактор для новичков
vim config.conf — мощный, но требует обучения
chmod — изменить права доступа к файлу/папке
chmod 755 script.sh — сделать файл исполняемым для владельца
chown — изменить владельца файла
sudo chown user:group file.txt
ps — показать запущенные процессы
ps aux — полный список процессов
top / htop — мониторинг системных ресурсов в реальном времени
(если htop не установлен: sudo apt install htop)
df — показать использование дискового пространства
df -h — в «человекочитаемом» виде (ГБ, МБ)
du — показать размер папки
du -sh /var/log — общий размер папки /var/log
grep — поиск по тексту
grep "error" /var/log/syslog — найти строки с «error»
grep -r "password" /etc/ — рекурсивный поиск в папке
find — найти файлы по имени, типу, размеру и т.д.
find /home -name "*.log" — найти все .log файлы в /home
sudo — выполнить команду от имени администратора (root)
sudo apt update
apt — менеджер пакетов в Ubuntu/Debian
sudo apt update — обновить список пакетов
sudo apt install nginx — установить пакет
sudo apt remove nginx — удалить пакет
systemctl — управление службами (сервисами)
sudo systemctl start nginx
sudo systemctl enable nginx — включить автозапуск
sudo systemctl status nginx — проверить статус
journalctl — просмотр системных логов (особенно для systemd-сервисов)
sudo journalctl -u nginx — логи конкретного сервиса
ping / curl / wget — сетевые утилиты
ping google.com — проверить доступность
curl http://localhost — получить ответ от веб-сервера
wget https://example.com/file.zip — скачать файл
